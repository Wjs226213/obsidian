#include <inttypes.h>  
#include <stdio.h>  
#include <time.h>  
#include <stdlib.h>  
  
int contains(int len, int arr[], int num);  
  
// 2分查找  
int main(int argc, char *argv[]) {  
    // 生成一个随机的数组并且要求随机的不相同  
    srand(time(NULL));  
    int arr[10] = {0};  
    int len = sizeof(arr) / sizeof(int);  
  
    for (int i = 0; i < len; i++) {  
        int num = rand() % 10 + 1;  
        // 判断当前的数字是不是和其他的数字重复  
  
        int result = contains(len, arr, num);  
        // 相同的数据我要再次完成生成如果不相同不需要执行操作  
  
        // 有相同数据  
        if (result) {  
            i--;  
        } else {  
            arr[i] = num;  
        }  
    }  
    for (int i = 0; i < len; ++i) {  
        printf("new %d data = %d \n", i, arr[i]);  
    }  
}  
  
int contains(int len, int arr[], int num) {  
    for (int i = 0; i < len; ++i) {  
        // 判断传输的随机数字是不是和列表中的数字相同  
        if (arr[i] == num) {  
            // 如果相同返回一个开关变量  
            return 1;  
        }  
    }  
    return 0; // 没有相同的变量  
}
